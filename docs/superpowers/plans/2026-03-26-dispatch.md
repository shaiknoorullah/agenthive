# Notification Dispatch & tmux Plugin Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the notification dispatch subsystem that routes agent notifications to multiple surfaces (tmux status bar, desktop notifications, Termux push, audio alerts) and the TPM plugin that renders notifications in tmux using native per-pane options with zero shell forks.

**Architecture:** A `Dispatcher` interface fans out notifications to platform-specific surfaces. The tmux surface writes per-pane user options (`@notif-msg`, `@notif-project`, `@notif-source`, `@notif-time`, `@notif-priority`) and the TPM plugin renders them via `#{W:#{P:#{?@notif-msg,...}}}` format strings built programmatically. Desktop surfaces use `notify-send` (Linux) or `osascript` (macOS) with action button support. Termux uses `termux-notification` with `--button1`/`--button2`. Audio uses terminal bell or system sounds. DND mode suppresses all surfaces except logging. Clearing is O(1) via an inline `pane-focus-in` hook.

**Tech Stack:** Go 1.22+, `github.com/stretchr/testify` (assertions), shell scripts (tmux plugin), `os/exec` (platform commands).

**Testing strategy:** TDD throughout. Unit tests with mock command executor verify dispatch logic without requiring tmux/notify-send. Integration tests verify tmux option writes against a real tmux server. Table-driven tests cover platform detection and format string generation. Race detector on all tests.

**Dependency:** Subsystem 2 (Core Daemon) provides the message router that calls `Dispatcher.Dispatch()`. This plan builds the dispatcher and all surfaces independently, tested via interfaces.

```go
package dispatch

import (
	"context"
	"errors"
	"fmt"
	"strings"
	"sync"
	"time"
)

// Priority represents notification urgency.
type Priority int

const (
	// PriorityInfo is the default notification priority.
	PriorityInfo Priority = iota
	// PriorityWarning indicates the agent needs attention.
	PriorityWarning
	// PriorityCritical indicates an error or urgent action required.
	PriorityCritical
)

var (
	// ErrEmptyMessage is returned when a notification has no message.
	ErrEmptyMessage = errors.New("notification message is required")
	// ErrEmptyPaneID is returned when a notification has no pane ID.
	ErrEmptyPaneID = errors.New("notification pane ID is required")
)

var priorityNames = map[Priority]string{
	PriorityInfo:     "info",
	PriorityWarning:  "warning",
	PriorityCritical: "critical",
}

// String returns the priority name.
func (p Priority) String() string {
	if name, ok := priorityNames[p]; ok {
		return name
	}
	return fmt.Sprintf("unknown(%d)", int(p))
}

// ParsePriority parses a priority string (case-insensitive).
func ParsePriority(s string) (Priority, error) {
	switch strings.ToLower(s) {
	case "info":
		return PriorityInfo, nil
	case "warning":
		return PriorityWarning, nil
	case "critical":
		return PriorityCritical, nil
	default:
		return 0, fmt.Errorf("unknown priority: %q", s)
	}
}

// Notification is a structured notification from an agent.
type Notification struct {
	Message  string    `json:"message"`
	Project  string    `json:"project"`
	Source   string    `json:"source"`
	Priority Priority  `json:"priority"`
	PaneID   string    `json:"pane_id"`
	Time     time.Time `json:"time"`
}

// Validate checks required fields and applies defaults.
// Mutates the notification in place to fill defaults.
func (n *Notification) Validate() error {
	if n.Message == "" {
		return ErrEmptyMessage
	}
	if n.PaneID == "" {
		return ErrEmptyPaneID
	}
	if n.Time.IsZero() {
		n.Time = time.Now()
	}
	// Priority defaults to PriorityInfo (zero value), which is correct.
	return nil
}

// Surface dispatches notifications to a specific target (tmux, desktop, etc.).
type Surface interface {
	// Name returns the surface identifier for logging and error reporting.
	Name() string
	// Dispatch sends a notification to this surface.
	Dispatch(ctx context.Context, n Notification) error
}

// Dispatcher fans out notifications to all registered surfaces.
// Safe for concurrent use.
type Dispatcher struct {
	mu       sync.RWMutex
	surfaces []Surface
	dnd      bool
}

// NewDispatcher creates a Dispatcher with the given surfaces.
func NewDispatcher(surfaces ...Surface) *Dispatcher {
	return &Dispatcher{
		surfaces: surfaces,
	}
}

// Dispatch validates the notification and fans it out to all surfaces.
// If DND mode is active, the notification is silently dropped (no error).
// If one or more surfaces fail, dispatch continues to remaining surfaces
// and returns a combined error.
func (d *Dispatcher) Dispatch(ctx context.Context, n Notification) error {
	if err := n.Validate(); err != nil {
		return fmt.Errorf("invalid notification: %w", err)
	}

	d.mu.RLock()
	dnd := d.dnd
	surfaces := d.surfaces
	d.mu.RUnlock()

	if dnd {
		return nil
	}

	var errs []error
	for _, s := range surfaces {
		if err := s.Dispatch(ctx, n); err != nil {
			errs = append(errs, fmt.Errorf("surface %s: %w", s.Name(), err))
		}
	}

	if len(errs) > 0 {
		return errors.Join(errs...)
	}
	return nil
}

// SetDND enables or disables Do Not Disturb mode.
func (d *Dispatcher) SetDND(on bool) {
	d.mu.Lock()
	defer d.mu.Unlock()
	d.dnd = on
}

// IsDND returns whether DND mode is active.
func (d *Dispatcher) IsDND() bool {
	d.mu.RLock()
	defer d.mu.RUnlock()
	return d.dnd
}
```

```go
package dispatch

import (
	"context"
	"os/exec"
	"sync"
)

// RecordedCmd is a command recorded by MockExecutor.
type RecordedCmd struct {
	Name string
	Args []string
}

// CmdExecutor abstracts command execution for testability.
type CmdExecutor interface {
	// Run executes a command and discards output.
	Run(ctx context.Context, name string, args ...string) error
	// Output executes a command and returns its stdout.
	Output(ctx context.Context, name string, args ...string) (string, error)
}

// RealExecutor runs real OS commands via os/exec.
type RealExecutor struct{}

// Run executes a command, discarding stdout and stderr.
func (e *RealExecutor) Run(ctx context.Context, name string, args ...string) error {
	return exec.CommandContext(ctx, name, args...).Run()
}

// Output executes a command and returns its stdout as a string.
func (e *RealExecutor) Output(ctx context.Context, name string, args ...string) (string, error) {
	out, err := exec.CommandContext(ctx, name, args...).Output()
	return string(out), err
}

// MockExecutor records commands for test assertions.
type MockExecutor struct {
	mu     sync.Mutex
	cmds   []RecordedCmd
	output string
	err    error
}

// NewMockExecutor creates a new mock executor.
func NewMockExecutor() *MockExecutor {
	return &MockExecutor{}
}

// Run records the command and returns the configured error.
func (m *MockExecutor) Run(ctx context.Context, name string, args ...string) error {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.cmds = append(m.cmds, RecordedCmd{Name: name, Args: args})
	return m.err
}

// Output records the command and returns the configured output and error.
func (m *MockExecutor) Output(ctx context.Context, name string, args ...string) (string, error) {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.cmds = append(m.cmds, RecordedCmd{Name: name, Args: args})
	return m.output, m.err
}

// SetOutput configures the output returned by Output().
func (m *MockExecutor) SetOutput(s string) {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.output = s
}

// SetError configures the error returned by Run() and Output().
func (m *MockExecutor) SetError(err error) {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.err = err
}

// Commands returns all recorded commands.
func (m *MockExecutor) Commands() []RecordedCmd {
	m.mu.Lock()
	defer m.mu.Unlock()
	cp := make([]RecordedCmd, len(m.cmds))
	copy(cp, m.cmds)
	return cp
}

// Reset clears all recorded commands and configuration.
func (m *MockExecutor) Reset() {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.cmds = nil
	m.output = ""
	m.err = nil
}
```

```go
package dispatch

import (
	"context"
	"fmt"
	"strings"
)

// Per-pane user option names. These are set on individual panes
// and read by the tmux format string in the status line.
const (
	optNotifMsg      = "@notif-msg"
	optNotifProject  = "@notif-project"
	optNotifSource   = "@notif-source"
	optNotifTime     = "@notif-time"
	optNotifPriority = "@notif-priority"
	optDND           = "@agenthive-dnd"
)

var allNotifOpts = []string{
	optNotifMsg,
	optNotifProject,
	optNotifSource,
	optNotifTime,
	optNotifPriority,
}

// TmuxSurface dispatches notifications by writing tmux per-pane user options.
// The tmux status line reads these options via format strings -- zero shell forks.
type TmuxSurface struct {
	exec CmdExecutor
}

// NewTmuxSurface creates a tmux surface with the given command executor.
func NewTmuxSurface(exec CmdExecutor) *TmuxSurface {
	return &TmuxSurface{exec: exec}
}

// Name returns the surface identifier.
func (s *TmuxSurface) Name() string {
	return "tmux"
}

// Dispatch sets per-pane user options for the notification's pane and
// refreshes all clients so the status line updates immediately.
func (s *TmuxSurface) Dispatch(ctx context.Context, n Notification) error {
	pane := n.PaneID
	timeStr := n.Time.Format("15:04:05")

	opts := []struct{ key, val string }{
		{optNotifMsg, n.Message},
		{optNotifProject, n.Project},
		{optNotifSource, n.Source},
		{optNotifTime, timeStr},
		{optNotifPriority, n.Priority.String()},
	}

	for _, o := range opts {
		if err := s.exec.Run(ctx, "tmux", "set-option", "-p", "-t", pane, o.key, o.val); err != nil {
			return fmt.Errorf("tmux set-option %s on pane %s: %w", o.key, pane, err)
		}
	}

	return s.refreshClients(ctx)
}

// Clear unsets all notification options on a pane and refreshes clients.
func (s *TmuxSurface) Clear(ctx context.Context, paneID string) error {
	for _, opt := range allNotifOpts {
		if err := s.exec.Run(ctx, "tmux", "set-option", "-p", "-u", "-t", paneID, opt); err != nil {
			return fmt.Errorf("tmux unset %s on pane %s: %w", opt, paneID, err)
		}
	}
	return s.refreshClients(ctx)
}

// SetDNDOption writes the global @agenthive-dnd tmux option.
func (s *TmuxSurface) SetDNDOption(ctx context.Context, on bool) error {
	val := "off"
	if on {
		val = "on"
	}
	return s.exec.Run(ctx, "tmux", "set-option", "-g", optDND, val)
}

// ListNotifications queries all panes for active notifications using
// tmux list-panes with a format string. Returns only panes that have
// a non-empty @notif-msg.
func (s *TmuxSurface) ListNotifications(ctx context.Context) ([]Notification, error) {
	format := "#{pane_id}|#{@notif-msg}|#{@notif-project}|#{@notif-source}|#{@notif-time}|#{@notif-priority}"
	out, err := s.exec.Output(ctx, "tmux", "list-panes", "-a", "-F", format)
	if err != nil {
		return nil, fmt.Errorf("tmux list-panes: %w", err)
	}

	var notifs []Notification
	for _, line := range strings.Split(out, "\n") {
		line = strings.TrimSpace(line)
		if line == "" {
			continue
		}

		parts := strings.SplitN(line, "|", 6)
		if len(parts) < 6 {
			continue
		}

		paneID := parts[0]
		msg := parts[1]
		if msg == "" {
			continue
		}

		priority, _ := ParsePriority(parts[5])

		notifs = append(notifs, Notification{
			PaneID:   paneID,
			Message:  msg,
			Project:  parts[2],
			Source:   parts[3],
			Priority: priority,
		})
	}

	return notifs, nil
}

func (s *TmuxSurface) refreshClients(ctx context.Context) error {
	return s.exec.Run(ctx, "tmux", "refresh-client", "-S")
}

// BuildNotificationFormatString returns the tmux format string for the status
// line that renders all pane notifications using zero shell forks.
//
// Uses #{W:#{P:...}} to iterate all windows and panes, checking each pane's
// @notif-msg option. Priority determines the color:
//   - critical: red
//   - warning:  yellow
//   - info:     default foreground
//
// The format string is built programmatically for readability. The raw string
// is dense but evaluates entirely inside the tmux server with no process forks.
func BuildNotificationFormatString() string {
	// Priority color selection:
	// #{?#{==:#{@notif-priority},critical},#[fg=red bold],#{?#{==:#{@notif-priority},warning},#[fg=yellow],#[default]}}
	colorOpen := "#{?#{==:#{@notif-priority},critical}," +
		"#[fg=red#{,}bold]," +
		"#{?#{==:#{@notif-priority},warning}," +
		"#[fg=yellow]," +
		"#[default]}}"

	// Notification content: [TIME] SOURCE/PROJECT: MSG
	content := "[#{@notif-time}] #{@notif-source}/#{@notif-project}: #{@notif-msg}"

	// Reset formatting after each notification
	colorClose := "#[default]  "

	// Per-pane conditional: only render if @notif-msg is set
	perPane := fmt.Sprintf("#{?#{@notif-msg},%s%s%s,}", colorOpen, content, colorClose)

	// Iterate all panes across all windows
	return fmt.Sprintf("#{W:#{P:%s}}", perPane)
}

// BuildClearHookCommand returns the tmux command string for the
// pane-focus-in hook that clears notifications in O(1).
//
// Uses if -F to evaluate a format string (no shell fork) and chains
// set -p -u commands to unset all notification options on the focused pane.
func BuildClearHookCommand() string {
	var unsets []string
	for _, opt := range allNotifOpts {
		unsets = append(unsets, fmt.Sprintf("set -p -u %s", opt))
	}
	chain := strings.Join(unsets, " ; ")
	return fmt.Sprintf(`if -F "#{@notif-msg}" "%s ; refresh-client -S"`, chain)
}
```

```go
package dispatch

import (
	"context"
	"fmt"
	"runtime"
)

// DesktopSurface dispatches notifications via OS-native desktop notifications.
// Linux: notify-send (standard) or notify-send.py (action buttons).
// macOS: osascript display notification (info) or display dialog (critical).
type DesktopSurface struct {
	exec            CmdExecutor
	platform        string
	hasNotifySendPy bool
}

// NewDesktopSurface creates a desktop notification surface.
// Auto-detects the platform on creation.
func NewDesktopSurface(exec CmdExecutor) *DesktopSurface {
	return &DesktopSurface{
		exec:     exec,
		platform: detectPlatform(),
	}
}

// Name returns the surface identifier.
func (s *DesktopSurface) Name() string {
	return "desktop"
}

// Dispatch sends a desktop notification appropriate to the platform.
func (s *DesktopSurface) Dispatch(ctx context.Context, n Notification) error {
	switch s.platform {
	case "linux":
		return s.dispatchLinux(ctx, n)
	case "darwin":
		return s.dispatchDarwin(ctx, n)
	default:
		return fmt.Errorf("unsupported platform for desktop notifications: %s", s.platform)
	}
}

func (s *DesktopSurface) dispatchLinux(ctx context.Context, n Notification) error {
	title := fmt.Sprintf("%s/%s", n.Source, n.Project)
	urgency := "normal"
	if n.Priority == PriorityCritical {
		urgency = "critical"
	}

	args := []string{
		title,
		n.Message,
		"--urgency", urgency,
	}

	return s.exec.Run(ctx, "notify-send", args...)
}

func (s *DesktopSurface) dispatchDarwin(ctx context.Context, n Notification) error {
	title := fmt.Sprintf("%s/%s", n.Source, n.Project)

	if n.Priority == PriorityCritical {
		// Critical: use display dialog with action buttons
		script := fmt.Sprintf(
			`display dialog %q with title %q buttons {"OK"} default button "OK" with icon caution giving up after 30`,
			n.Message, title,
		)
		return s.exec.Run(ctx, "osascript", "-e", script)
	}

	// Info/Warning: use display notification (non-blocking banner)
	script := fmt.Sprintf(
		`display notification %q with title %q`,
		n.Message, title,
	)
	return s.exec.Run(ctx, "osascript", "-e", script)
}

// detectPlatform returns the current OS.
func detectPlatform() string {
	return runtime.GOOS
}
```

```go
import (
	"context"
```

```go
// AudioMode determines the audio alert behavior.
type AudioMode int

const (
	// AudioModeOff disables audio alerts.
	AudioModeOff AudioMode = iota
	// AudioModeBell plays the terminal bell (BEL character).
	AudioModeBell
	// AudioModeSystem plays the default system notification sound.
	AudioModeSystem
	// AudioModeCustom plays a user-specified sound file.
	AudioModeCustom
)

// Default system sound paths.
const (
	darwinDefaultSound = "/System/Library/Sounds/Tink.aiff"
	linuxDefaultSound  = "/usr/share/sounds/freedesktop/stereo/complete.oga"
)

// AudioSurface plays audio alerts for notifications.
type AudioSurface struct {
	exec       CmdExecutor
	mode       AudioMode
	platform   string
	customPath string
}

// NewAudioSurface creates an audio surface with the given mode.
func NewAudioSurface(exec CmdExecutor, mode AudioMode) *AudioSurface {
	return &AudioSurface{
		exec:     exec,
		mode:     mode,
		platform: detectPlatform(),
	}
}

// Name returns the surface identifier.
func (s *AudioSurface) Name() string {
	return "audio"
}

// Dispatch plays an audio alert based on the configured mode.
func (s *AudioSurface) Dispatch(ctx context.Context, n Notification) error {
	switch s.mode {
	case AudioModeOff:
		return nil

	case AudioModeBell:
		return s.exec.Run(ctx, "printf", `\a`)

	case AudioModeSystem:
		return s.playSound(ctx, s.defaultSoundPath())

	case AudioModeCustom:
		return s.playSound(ctx, s.customPath)

	default:
		return fmt.Errorf("unknown audio mode: %d", s.mode)
	}
}

func (s *AudioSurface) playSound(ctx context.Context, path string) error {
	switch s.platform {
	case "darwin":
		return s.exec.Run(ctx, "afplay", path)
	case "linux":
		return s.exec.Run(ctx, "paplay", path)
	default:
		return fmt.Errorf("unsupported platform for system audio: %s", s.platform)
	}
}

func (s *AudioSurface) defaultSoundPath() string {
	switch s.platform {
	case "darwin":
		return darwinDefaultSound
	case "linux":
		return linuxDefaultSound
	default:
		return ""
	}
}

// ParseAudioMode parses a user-provided audio mode string.
// Recognized values: "off", "bell", "system".
// Any other non-empty value is treated as a custom sound file path.
func ParseAudioMode(s string) AudioMode {
	switch strings.ToLower(s) {
	case "", "off":
		return AudioModeOff
	case "bell":
		return AudioModeBell
	case "system":
		return AudioModeSystem
	default:
		return AudioModeCustom
	}
}
```

The TPM plugin entry point is a shell script that registers keybindings, builds the notification format string for the status line, installs the pane-focus-in clearing hook, and reads user configuration from tmux options. This is the bridge between the Go daemon and tmux.

- [ ] **Step 1: Write the TPM plugin script**

Create `tmux/agenthive.tmux`:

```bash
#!/usr/bin/env bash
# agenthive.tmux - TPM plugin entry point
#
# Installs notification rendering into the tmux status line using native
# per-pane user options. Zero shell forks for rendering.
#
# Per-pane options used:
#   @notif-msg       - Notification message text (clean, no format codes)
#   @notif-project   - Project name
#   @notif-source    - Source tool (Claude, Codex, etc.)
#   @notif-time      - Timestamp (HH:MM:SS)
#   @notif-priority  - Priority level (info, warning, critical)
#
# Global options read:
#   @agenthive-key-next       - Keybinding for jump-to-oldest (default: N)
#   @agenthive-key-picker     - Keybinding for fzf picker (default: S)
#   @agenthive-key-dashboard  - Keybinding for dashboard popup (default: D)
#   @agenthive-status-line    - Enable status line integration (default: on)
#   @agenthive-dnd            - Do Not Disturb mode (default: off)

CURRENT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

# -- Helpers --

tmux_option() {
    local option="$1"
    local default_value="$2"
    local value
    value=$(tmux show-option -gqv "$option")
    echo "${value:-$default_value}"
}

# -- Version Check --

tmux_version=$(tmux -V | grep -oE '[0-9]+\.[0-9]+')
required="3.2"
if printf '%s\n%s\n' "$required" "$tmux_version" | sort -V | head -1 | grep -qv "$required"; then
    tmux display-message "agenthive: tmux $required+ required (found $tmux_version)"
    exit 1
fi

# -- Read User Configuration --

key_next=$(tmux_option "@agenthive-key-next" "N")
key_picker=$(tmux_option "@agenthive-key-picker" "S")
key_dashboard=$(tmux_option "@agenthive-key-dashboard" "D")
key_dnd=$(tmux_option "@agenthive-key-dnd" "Q")
status_line=$(tmux_option "@agenthive-status-line" "on")

# -- Build Notification Format String --
#
# This format string is evaluated entirely inside the tmux server.
# #{W:#{P:...}} iterates all windows and panes.
# #{?@notif-msg,...,} only renders for panes with an active notification.
# Priority colors: critical=red, warning=yellow, info=default.
#
# Built programmatically for readability. The result is a single format
# string passed to tmux set-option.

build_notification_format() {
    local color_open priority_check content color_close per_pane

    # Priority-based color selection (nested conditionals)
    priority_check='#{?#{==:#{@notif-priority},critical},'
    priority_check+='#[fg=red#,bold],'
    priority_check+='#{?#{==:#{@notif-priority},warning},'
    priority_check+='#[fg=yellow],'
    priority_check+='#[default]}}'

    # Notification content
    content='[#{@notif-time}] #{@notif-source}/#{@notif-project}: #{@notif-msg}'

    # Reset after each notification
    color_close='#[default]  '

    # Per-pane conditional
    per_pane="#{?#{@notif-msg},${priority_check}${content}${color_close},}"

    # Iterate all panes across all windows
    echo "#{W:#{P:${per_pane}}}"
}

# -- Install Status Line --

if [ "$status_line" = "on" ]; then
    notif_format=$(build_notification_format)

    # Append notification format to the right status line
    current_right=$(tmux show-option -gqv "status-right")

    # Only add if not already present (idempotent)
    if [[ "$current_right" != *"@notif-msg"* ]]; then
        tmux set-option -g status-right "${notif_format}${current_right}"
    fi
fi

# -- Install Clearing Hook --
#
# On pane focus, clear notifications for the focused pane.
# Uses if -F (format evaluation, no fork) for O(1) clearing.

clear_cmd='if -F "#{@notif-msg}" "'
clear_cmd+='set -p -u @notif-msg ; '
clear_cmd+='set -p -u @notif-project ; '
clear_cmd+='set -p -u @notif-source ; '
clear_cmd+='set -p -u @notif-time ; '
clear_cmd+='set -p -u @notif-priority ; '
clear_cmd+='refresh-client -S"'

tmux set-hook -g pane-focus-in "$clear_cmd"

# -- Keybindings --

# Jump to oldest notification pane
tmux bind-key "$key_next" run-shell "$CURRENT_DIR/scripts/notification-picker.sh --jump-oldest"

# Open fzf notification picker
tmux bind-key "$key_picker" display-popup -w 80% -h 60% -E "$CURRENT_DIR/scripts/notification-picker.sh"

# Dashboard popup
tmux bind-key "$key_dashboard" display-popup -w 90% -h 80% -E "$CURRENT_DIR/scripts/notification-picker.sh --dashboard"

# Toggle DND
tmux bind-key "$key_dnd" run-shell "
    dnd=\$(tmux show-option -gqv @agenthive-dnd);
    if [ \"\$dnd\" = 'on' ]; then
        tmux set-option -g @agenthive-dnd off;
        tmux display-message 'agenthive: DND off';
    else
        tmux set-option -g @agenthive-dnd on;
        tmux display-message 'agenthive: DND on';
    fi
"
```

## Task 14: Notification Picker Script

**Files:**

The notification picker reads active notifications from tmux per-pane options via `tmux list-panes -a` and presents them in fzf for selection. Selecting a notification jumps to that pane. The `--jump-oldest` flag skips fzf and jumps directly to the oldest notification pane.

- [ ] **Step 1: Write the picker script**

Create `tmux/scripts/notification-picker.sh`:

```bash
#!/usr/bin/env bash
# notification-picker.sh - fzf-based notification picker using tmux per-pane options
#
# Modes:
#   (no args)       - fzf picker for all active notifications
#   --jump-oldest   - jump directly to the oldest notification pane
#   --dashboard     - show all agents with state and notification info
#
# Data source: tmux list-panes -a with per-pane @notif-* options.
# No files are read. All data comes from the tmux server.

set -euo pipefail

MODE="${1:-picker}"

# -- Helpers --

# List all panes with active notifications.
# Output: PANE_ID<tab>TIME<tab>SOURCE/PROJECT: MSG<tab>PRIORITY
list_notifications() {
    tmux list-panes -a -F \
        '#{pane_id}	#{@notif-time}	#{@notif-source}/#{@notif-project}: #{@notif-msg}	#{@notif-priority}' \
    | while IFS=$'\t' read -r pane_id time_str content priority; do
        # Skip panes with no notification
        if [ -n "$content" ] && [ "$content" != "/: " ] && [ "$content" != ": " ]; then
            echo -e "${pane_id}\t${time_str}\t${content}\t${priority}"
        fi
    done | sort -t$'\t' -k2,2
}

# Jump to a pane by its ID.
jump_to_pane() {
    local pane_id="$1"
    # Find the window containing this pane and switch to it
    local window_id
    window_id=$(tmux list-panes -a -F '#{pane_id} #{window_id}' | grep "^${pane_id} " | head -1 | awk '{print $2}')

    if [ -n "$window_id" ]; then
        tmux select-window -t "$window_id"
        tmux select-pane -t "$pane_id"
    fi
}

# Color a line based on priority for fzf display.
color_line() {
    local line="$1"
    local priority
    priority=$(echo "$line" | awk -F'\t' '{print $4}')

    case "$priority" in
        critical) echo -e "\033[1;31m${line}\033[0m" ;;
        warning)  echo -e "\033[33m${line}\033[0m" ;;
        *)        echo "$line" ;;
    esac
}

# -- Modes --

case "$MODE" in
    --jump-oldest)
        # Jump to the oldest (first) notification pane
        oldest=$(list_notifications | head -1)
        if [ -z "$oldest" ]; then
            tmux display-message "agenthive: no active notifications"
            exit 0
        fi
        pane_id=$(echo "$oldest" | cut -f1)
        jump_to_pane "$pane_id"
        ;;

    --dashboard)
        # Show all panes with their agent state and notifications
        echo "=== agenthive Dashboard ==="
        echo ""
        printf "%-8s %-20s %-10s %-10s %s\n" "PANE" "PROJECT" "SOURCE" "PRIORITY" "MESSAGE"
        echo "-------- -------------------- ---------- ---------- -------"

        list_notifications | while IFS=$'\t' read -r pane_id time_str content priority; do
            project=$(echo "$content" | cut -d: -f1 | xargs)
            msg=$(echo "$content" | cut -d: -f2- | xargs)
            source=$(echo "$project" | cut -d/ -f1)
            proj=$(echo "$project" | cut -d/ -f2-)
            printf "%-8s %-20s %-10s %-10s [%s] %s\n" "$pane_id" "$proj" "$source" "$priority" "$time_str" "$msg"
        done

        echo ""
        echo "Press q to close"
        read -n 1 -s key
        ;;

    picker|*)
        # fzf picker
        notifications=$(list_notifications)

        if [ -z "$notifications" ]; then
            tmux display-message "agenthive: no active notifications"
            exit 0
        fi

        # Format for display: [TIME] SOURCE/PROJECT: MSG (PRIORITY)
        display=$(echo "$notifications" | while IFS=$'\t' read -r pane_id time_str content priority; do
            echo "${pane_id}	[${time_str}] ${content}"
        done)

        # Run fzf and extract pane ID from selection
        selected=$(echo "$display" | fzf \
            --ansi \
            --no-preview \
            --header="Select notification (Enter to jump, Esc to cancel)" \
            --delimiter=$'\t' \
            --with-nth=2 \
            --no-sort \
        ) || exit 0

        pane_id=$(echo "$selected" | cut -f1)
        if [ -n "$pane_id" ]; then
            jump_to_pane "$pane_id"
        fi
        ;;
esac
```
